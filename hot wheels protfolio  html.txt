    <script>
        // Data Storage (LocalStorage)
        const Storage = {
            getUsers: () => JSON.parse(localStorage.getItem('mp_users') || '[]'),
            saveUsers: (users) => localStorage.setItem('mp_users', JSON.stringify(users)),
            getProducts: () => JSON.parse(localStorage.getItem('mp_products') || '[]'),
            saveProducts: (products) => localStorage.setItem('mp_products', JSON.stringify(products)),
            getCurrentUser: () => JSON.parse(localStorage.getItem('mp_current_user') || 'null'),
            setCurrentUser: (user) => localStorage.setItem('mp_current_user', JSON.stringify(user))
        };

        // Initialize demo data if empty
        function initDemoData() {
            if (Storage.getUsers().length === 0) {
                const demoUsers = [
                    { id: 1, name: 'Alice Johnson', email: 'alice@shop.com', password: '123456', business: 'Alice Fashion Hub' },
                    { id: 2, name: 'Bob Smith', email: 'bob@tech.com', password: '123456', business: 'Bob Tech Store' }
                ];
                const demoProducts = [
                    { id: 1, sellerId: 1, sellerName: 'Alice Johnson', sellerBusiness: 'Alice Fashion Hub', name: 'Designer Handbag', price: 89.99, category: 'fashion', condition: 'new', desc: 'Premium leather handbag, perfect for daily use.', date: '2026-04-28' },
                    { id: 2, sellerId: 2, sellerName: 'Bob Smith', sellerBusiness: 'Bob Tech Store', name: 'Bluetooth Speaker', price: 45.00, category: 'electronics', condition: 'new', desc: 'Portable wireless speaker with amazing bass.', date: '2026-04-29' },
                    { id: 3, sellerId: 1, sellerName: 'Alice Johnson', sellerBusiness: 'Alice Fashion Hub', name: 'Summer Dress', price: 35.50, category: 'fashion', condition: 'new', desc: 'Light and breezy summer dress in floral print.', date: '2026-04-30' }
                ];
                Storage.saveUsers(demoUsers);
                Storage.saveProducts(demoProducts);
            }
        }

        // Page Navigation
        function showPage(page) {
            // Hide all pages
            document.querySelectorAll('.auth-container, .container').forEach(el => el.classList.add('hidden'));
            
            // Show requested page
            if (page === 'login') document.getElementById('loginPage').classList.remove('hidden');
            else if (page === 'register') document.getElementById('registerPage').classList.remove('hidden');
            else if (page === 'home') {
                document.getElementById('homePage').classList.remove('hidden');
                renderAllProducts();
            }
            else if (page === 'sellers') {
                document.getElementById('sellersPage').classList.remove('hidden');
                renderSellers();
            }
            else if (page === 'myProfile') {
                if (!Storage.getCurrentUser()) { showPage('login'); return; }
                document.getElementById('profilePage').classList.remove('hidden');
                renderMyProfile();
            }
            else if (page === 'sellerPublic') {
                document.getElementById('sellerPublicPage').classList.remove('hidden');
            }
            
            // Update nav
            updateNav();
        }

        function updateNav() {
            const user = Storage.getCurrentUser();
            if (user) {
                document.getElementById('userNav').classList.remove('hidden');
                document.getElementById('guestNav').classList.add('hidden');
                document.getElementById('loggedInUser').textContent = `Hi, ${user.name}`;
            } else {
                document.getElementById('userNav').classList.add('hidden');
                document.getElementById('guestNav').classList.remove('hidden');
            }
        }

        // Registration
        document.getElementById('registerForm').addEventListener('submit', function(e) {
            e.preventDefault();
            const users = Storage.getUsers();
            const email = document.getElementById('regEmail').value;
            
            if (users.find(u => u.email === email)) {
                document.getElementById('regError').style.display = 'block';
                return;
            }
            
            const newUser = {
                id: Date.now(),
                name: document.getElementById('regName').value,
                email: email,
                password: document.getElementById('regPassword').value,
                business: document.getElementById('regBusiness').value
            };
            
            users.push(newUser);
            Storage.saveUsers(users);
            Storage.setCurrentUser(newUser);
            showPage('myProfile');
        });

        // Login
        document.getElementById('loginForm').addEventListener('submit', function(e) {
            e.preventDefault();
            const users = Storage.getUsers();
            const email = document.getElementById('loginEmail').value;
            const password = document.getElementById('loginPassword').value;
            
            const user = users.find(u => u.email === email && u.password === password);
            
            if (user) {
                Storage.setCurrentUser(user);
                showPage('myProfile');
            } else {
                document.getElementById('loginError').style.display = 'block';
            }
        });

        // Logout
        function logout() {
            Storage.setCurrentUser(null);
            showPage('home');
        }

        // Add Product
        document.getElementById('addProductForm').addEventListener('submit', function(e) {
            e.preventDefault();
            const user = Storage.getCurrentUser();
            if (!user) return;
            
            const products = Storage.getProducts();
            const newProduct = {
                id: Date.now(),
                sellerId: user.id,
                sellerName: user.name,
                sellerBusiness: user.business,
                name: document.getElementById('prodName').value,
                price: parseFloat(document.getElementById('prodPrice').value),
                category: document.getElementById('prodCategory').value,
                condition: document.getElementById('prodCondition').value,
                desc: document.getElementById('prodDesc').value,
                date: new Date().toISOString().split('T')[0]
            };
            
            products.push(newProduct);
            Storage.saveProducts(products);
            this.reset();
            renderMyProfile();
            alert('✅ Product posted successfully!');
        });

        // Render Functions
        function renderAllProducts(category = 'all') {
            const products = Storage.getProducts();
            const filtered = category === 'all' ? products : products.filter(p => p.category === category);
            const grid = document.getElementById('allProductsGrid');
            
            grid.innerHTML = filtered.length === 0 ? '<p style="grid-column: 1/-1; text-align: center; color: #6b7280;">No products found.</p>' : 
                filtered.map(p => `
                    <div class="product-card">
                        <div class="product-image">📦</div>
                        <div class="product-details">
                            <div class="product-seller" onclick="viewSellerProfile(${p.sellerId})" style="cursor: pointer;">🏪 ${p.sellerBusiness}</div>
                            <div class="product-title">${p.name}</div>
                            <div class="product-desc">${p.desc}</div>
                            <div style="font-size: 0.875rem; color: #6b7280; margin-bottom: 0.5rem;">
                                <span style="background: #dbeafe; color: #1e40af; padding: 0.25rem 0.5rem; border-radius: 4px;">${p.condition}</span>
                                <span style="background: #fce7f3; color: #be185d; padding: 0.25rem 0.5rem; border-radius: 4px; margin-left: 0.5rem;">${p.category}</span>
                            </div>
                            <div class="product-footer">
                                <span class="product-price">$${p.price.toFixed(2)}</span>
                                <span style="color: #9ca3af; font-size: 0.875rem;">${p.date}</span>
                            </div>
                        </div>
                    </div>
                `).join('');
        }

        function filterProducts(category, btn) {
            document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
            btn.classList.add('active');
            renderAllProducts(category);
        }

        function renderSellers() {
            const users = Storage.getUsers();
            const products = Storage.getProducts();
            const grid = document.getElementById('sellersGrid');
            
            grid.innerHTML = users.map(u => {
                const count = products.filter(p => p.sellerId === u.id).length;
                return `
                    <div class="seller-card" onclick="viewSellerProfile(${u.id})">
                        <div class="seller-avatar">${u.name.split(' ').map(n => n[0]).join('')}</div>
                        <div class="seller-name">${u.name}</div>
                        <div style="color: #3b82f6; font-size: 0.875rem; margin-bottom: 0.25rem;">${u.business}</div>
                        <div class="seller-count">${count} product${count !== 1 ? 's' : ''} listed</div>
                    </div>
                `;
            }).join('');
        }

        function renderMyProfile() {
            const user = Storage.getCurrentUser();
            if (!user) return;
            
            document.getElementById('profileName').textContent = user.name;
            document.getElementById('profileBusiness').textContent = user.business;
            document.getElementById('profileEmail').textContent = user.email;
            document.getElementById('profileAvatar').textContent = user.name.split(' ').map(n => n[0]).join('');
            
            const products = Storage.getProducts().filter(p => p.sellerId === user.id);
            const grid = document.getElementById('myProductsGrid');
            
            grid.innerHTML = products.length === 0 ? '<p style="grid-column: 1/-1; text-align: center; color: #6b7280;">You haven\'t posted any products yet.</p>' :
                products.map(p => `
                    <div class="product-card">
                        <div class="product-image">📦</div>
                        <div class="product-details">
                            <div class="product-title">${p.name}</div>
                            <div class="product-desc">${p.desc}</div>
                            <div style="font-size: 0.875rem; color: #6b7280; margin-bottom: 0.5rem;">
                                <span style="background: #dbeafe; color: #1e40af; padding: 0.25rem 0.5rem; border-radius: 4px;">${p.condition}</span>
                            </div>
                            <div class="product-footer">
                                <span class="product-price">$${p.price.toFixed(2)}</span>
                                <button class="delete-btn" onclick="deleteProduct(${p.id})">Delete</button>
                            </div>
                        </div>
                    </div>
                `).join('');
        }

        function viewSellerProfile(sellerId) {
            const users = Storage.getUsers();
            const products = Storage.getProducts();
            const seller = users.find(u => u.id === sellerId);
            
            if (!seller) return;
            
            document.getElementById('publicName').textContent = seller.name;
            document.getElementById('publicBusiness').textContent = seller.business;
            document.getElementById('publicAvatar').textContent = seller.name.split(' ').map(n => n[0]).join('');
            
            const sellerProducts = products.filter(p => p.sellerId === sellerId);
            const grid = document.getElementById('publicProductsGrid');
            
            grid.innerHTML = sellerProducts.length === 0 ? '<p style="grid-column: 1/-1; text-align: center; color: #6b7280;">No products yet.</p>' :
                sellerProducts.map(p => `
                    <div class="product-card">
                        <div class="product-image">📦</div>
                        <div class="product-details">
                            <div class="product-title">${p.name}</div>
                            <div class="product-desc">${p.desc}</div>
                            <div class="product-footer">
                                <span class="product-price">$${p.price.toFixed(2)}</span>
                            </div>
                        </div>
                    </div>
                `).join('');
            
            showPage('sellerPublic');
        }

        function deleteProduct(productId) {
            if (!confirm('Are you sure you want to delete this product?')) return;
            const products = Storage.getProducts().filter(p => p.id !== productId);
            Storage.saveProducts(products);
            renderMyProfile();
        }

        // Initialize
        initDemoData();
        showPage('home');
    </script>
